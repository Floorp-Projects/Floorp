BEGIN:VEVENT
DESCRIPTION
 :The 3rd instance into the month of one of Tuesday\, Wednesday or 
 Thursday\, for the next 6 months:
DTSTART
 ;TZID=US-Eastern
 :19970904T090000
RRULE
 :FREQ=MONTHLY;COUNT=6;INTERVAL=1;BYDAY=TU,WE,TH;BYSETPOS=3
END:VEVENT

BEGIN:VEVENT
DESCRIPTION

END:VEVENT
